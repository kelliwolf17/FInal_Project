Select cg.State, t1.`Restaurant_Total`, cg.Total AS `Gym_Total`
from clean_gym cg
         join (Select province, count(*) AS `Restaurant_Total` from clean_restaurant cr group by 1) t1
              on cg.State = t1.province
group by 1;
